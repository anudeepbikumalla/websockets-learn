// Map of userId ‚Üí { ws, username, joinedAt }
const users = new Map();

wss.on('connection', (ws) => {
  ws.on('message', (raw) => {
    const msg = JSON.parse(raw);

    if (msg.type === 'join') {
      ws.userId = msg.userId;
      users.set(msg.userId, { ws, username: msg.username, joinedAt: Date.now() });
      broadcastPresence(); // tell everyone
    }
  });

  ws.on('close', () => {
    users.delete(ws.userId);
    broadcastPresence(); // tell everyone they left
  });
});

function broadcastPresence() {
  const list = [...users.values()].map(u => u.username);
  const msg = JSON.stringify({ type: 'presence', users: list });
  wss.clients.forEach(c => c.readyState === 1 && c.send(msg));
}

‚úÖ Presence Rules

üó∫Ô∏è
Use a Map for the registry ‚Äî key by userId. Map gives O(1) lookup for join/leave and keeps insertion order.
üîå
Delete on disconnect ‚Äî always call users.delete(ws.userId) in the close handler. Connections can drop without announcement.
üì§
Send full list, not diffs ‚Äî sending the complete presence list on every change is simpler and more reliable than sending incremental updates.
‚è±Ô∏è
Debounce rapid joins ‚Äî if 50 users join in 1 second, debounce broadcastPresence() by 200ms to avoid 50 broadcasts.

Document Co-Editing Presence
const activeEditors = new Map(); // ws -> { name, cursorPosition }
wss.on('connection', (ws) => {
  // Broadast when someone joins the document
  broadcastPresence(activeEditors);
  ws.on('message', (msg) => {
    const data = JSON.parse(msg);
    if (data.type === 'CURSOR_MOVE') {
      activeEditors.get(ws).cursorPosition = data.pos;
      broadcastPresence(activeEditors);
    }
  });
  ws.on('close', () => {
    activeEditors.delete(ws);
    broadcastPresence(activeEditors); // Instantly update active viewers
  });
});

üí° Key Takeaways for Beginners
‚ñ∏A presence system answers one question: "Who is online right now?" ‚Äî like the green dots in Slack.
‚ñ∏Use a Map to track online users: userId ‚Üí WebSocket connection. Add on connect, delete on disconnect.
‚ñ∏When someone joins or leaves, broadcast the updated user list to everyone. This keeps all clients in sync.
‚ñ∏Handle the close event carefully ‚Äî a user might have multiple tabs open. Only mark offline when ALL tabs close.
‚ñ∏This pattern is used in Slack, Discord, Zoom, Google Docs ‚Äî anywhere you see "5 people online" or typing indicators.
üíª Server-side presence tracking
const onlineUsers = new Map(); // userId ‚Üí ws

wss.on('connection', (ws, req) => {
  const userId = authenticate(req); // Get user from token

  // Mark user as online
  onlineUsers.set(userId, ws);
  broadcastPresence(); // Tell everyone

  ws.on('close', () => {
    // Mark user as offline
    onlineUsers.delete(userId);
    broadcastPresence(); // Tell everyone
  });
});

function broadcastPresence() {
  const userList = Array.from(onlineUsers.keys());
  const msg = JSON.stringify({
    type: 'presence',
    users: userList,        // ['alice', 'bob', 'charlie']
    count: userList.length  // 3
  });

  wss.clients.forEach((client) => {
    if (client.readyState === WebSocket.OPEN) {
      client.send(msg);
    }
  });
}
