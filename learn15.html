// Map of userId ‚Üí { ws, username, joinedAt }
const users = new Map();

wss.on('connection', (ws) => {
  ws.on('message', (raw) => {
    const msg = JSON.parse(raw);

    if (msg.type === 'join') {
      ws.userId = msg.userId;
      users.set(msg.userId, { ws, username: msg.username, joinedAt: Date.now() });
      broadcastPresence(); // tell everyone
    }
  });

  ws.on('close', () => {
    users.delete(ws.userId);
    broadcastPresence(); // tell everyone they left
  });
});

function broadcastPresence() {
  const list = [...users.values()].map(u => u.username);
  const msg = JSON.stringify({ type: 'presence', users: list });
  wss.clients.forEach(c => c.readyState === 1 && c.send(msg));
}

‚úÖ Presence Rules

üó∫Ô∏è
Use a Map for the registry ‚Äî key by userId. Map gives O(1) lookup for join/leave and keeps insertion order.
üîå
Delete on disconnect ‚Äî always call users.delete(ws.userId) in the close handler. Connections can drop without announcement.
üì§
Send full list, not diffs ‚Äî sending the complete presence list on every change is simpler and more reliable than sending incremental updates.
‚è±Ô∏è
Debounce rapid joins ‚Äî if 50 users join in 1 second, debounce broadcastPresence() by 200ms to avoid 50 broadcasts.
