Token Bucket Algorithm ü™£
The industry-standard rate limiting algorithm
</> server.js ‚Äî Token Bucket per client
// Each client gets their own bucket
function createBucket(capacity = 10, refillRate = 10) {
  return {
    tokens: capacity,    // starts full
    capacity,
    refillRate,       // tokens added per second
    lastRefill: Date.now()
  };
}

function consume(bucket) {
  // Refill tokens based on elapsed time
  const now = Date.now();
  const elapsed = (now - bucket.lastRefill) / 1000;
  bucket.tokens = Math.min(bucket.capacity, bucket.tokens + elapsed * bucket.refillRate);
  bucket.lastRefill = now;

  if (bucket.tokens >= 1) {
    bucket.tokens--;
    return true;  // ‚úÖ allowed
  }
  return false;  // ‚ùå rate limited
}

wss.on('connection', (ws) => {
  ws.bucket = createBucket(10, 10); // 10 tokens, refills 10/sec
  ws.on('message', (msg) => {
    if (!consume(ws.bucket)) {
      ws.send(JSON.stringify({ type: 'error', msg: 'Rate limit exceeded' }));
      return;
    }
    handleMessage(msg);
  });
});

‚ö° Rate Limiting Rules
Production checklist

‚úÖ Rate Limiting Checklist

ü™£
Use Token Bucket ‚Äî it allows short bursts (natural user behavior) while blocking sustained spam.
üë§
Rate limit per client ‚Äî attach a bucket to each WebSocket object (ws.bucket). Never share buckets between clients.
üì§
Send a warning first ‚Äî on the first rate limit hit, send an error message. Close the connection only on repeated/severe violations.
üî¢
Use custom close code 4029 ‚Äî the HTTP "429 Too Many Requests" equivalent for WebSockets.
üìè
Also limit message size ‚Äî check msg.length and drop oversized payloads (>64KB for chat, more for file uploads).

üí° Key Takeaways for Beginners
‚ñ∏Without rate limiting, a single malicious client can send thousands of messages per second and crash your server.
‚ñ∏The Token Bucket algorithm: You have N tokens. Each message costs 1 token. Tokens refill over time. No tokens = message rejected.
‚ñ∏This is like a coffee punch card ‚Äî you get 10 free coffees, use them up, and have to wait for new ones.
‚ñ∏Set reasonable limits: e.g., 10 messages per second for chat, 1 message per second for expensive operations.
‚ñ∏Always warn the user before disconnecting them. Send a warning at 80% usage, disconnect at 100%.
üíª Token Bucket rate limiter
class TokenBucket {
  constructor(maxTokens, refillRate) {
    this.tokens = maxTokens;     // Start full
    this.maxTokens = maxTokens;  // Maximum capacity
    this.refillRate = refillRate; // Tokens added per second

    // Refill tokens over time
    setInterval(() => {
      this.tokens = Math.min(this.maxTokens, this.tokens + 1);
    }, 1000 / this.refillRate);
  }

  // Returns true if message is allowed, false if rate limited
  consume() {
    if (this.tokens > 0) {
      this.tokens--;
      return true;   // ‚úÖ Allowed
    }
    return false;     // ‚ùå Rate limited!
  }
}

// Give each client their own bucket (10 msgs max, refill 2/sec)
wss.on('connection', (ws) => {
  const bucket = new TokenBucket(10, 2);

  ws.on('message', (data) => {
    if (!bucket.consume()) {
      ws.send(JSON.stringify({ error: '‚ö†Ô∏è Slow down! Rate limited.' }));
      return; // Drop the message
    }
    // Process the message normally...
  });
});
