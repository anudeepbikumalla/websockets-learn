const metrics = {
  connections: 0,    // current active connections
  totalConn: 0,     // lifetime total connections
  messages: 0,     // total messages received
  errors: 0,       // total errors
  bytes: 0         // total bytes received
};

wss.on('connection', (ws) => {
  metrics.connections++; metrics.totalConn++;
  console.log(JSON.stringify({ event: 'connect', connections: metrics.connections }));

  ws.on('message', (msg) => { metrics.messages++; metrics.bytes += msg.length; });
  ws.on('error', () => metrics.errors++);
  ws.on('close', () => { metrics.connections--; console.log(JSON.stringify({ event: 'disconnect', connections: metrics.connections })); });
});

// Expose via HTTP endpoint (for Prometheus, Datadog, etc.)
app.get('/metrics', (req, res) => res.json(metrics));

‚úÖ Monitoring Rules

üìù
Log as JSON ‚Äî structured logs (JSON) can be parsed by Datadog, Grafana, CloudWatch. Human-readable logs can't be queried.
üö®
Alert on connection drops ‚Äî set up an alert if active connections drop by >20% in 1 minute. Could be a server crash or deploy issue.
üì°
Expose a /metrics endpoint ‚Äî Prometheus scrapes it. Grafana visualizes it. This is the industry standard for production monitoring.
üîá
Don't log ping/pong ‚Äî they fire every 30 seconds and will flood your logs. Filter them out in your logging middleware.
